\begin{PROBLEM}
    \p
    تعداد جواب‌های معادله زیر را برحسب
    $n$ و $k$
    در مجموعه اعداد حسابی بیابید:
    $$\sum_{i=1}^{n} x_i = k$$
    $$\forall i \in \mathbb{N}: x_i \ge i$$

    \SOLUTION[پاسخ به روش توزیع اشیا]{
        \p
        تغییر متغیر زیر را درنظر می‌گیریم:
        $$\forall i \in \mathbb{N}: y_i = x_i-i$$
        بنابراین، می‌توانیم معادله‌ی داده شده را بازنویسی کنیم:
        $$\sum_{i=1}^{n} x_i = \sum_{i=1}^n (y_i+i) = \sum_{i=1}^n y_i + \frac{n(n+1)}{2} = k$$
        $$\sum_{i=1}^n y_i = k - \frac{n(n+1)}{2}$$
        که در آن:
        $$\forall i \in \mathbb{N}: y_i \ge i-i=0$$
        \CROSSREF[می‌دانیم]{تعداد پاسخ‌های معادله سیاله خطی با ضرایب واحد در مجموعه اعداد حسابی}
        تعداد پاسخ‌های معادله‌ی بالا برابر است با:
        $${k - \frac{n(n+1)}{2} + n - 1 \choose n - 1}$$
    }

    \SOLUTION[پاسخ به روش تابع مولد]{
        \p
        با نام‌گذاری زیر:
        $$\sum_{i=1}^\infty y^i = (y+y^2+...) = Y$$
        تابع مولد معادله‌ی داده شده برابر است با
        \begin{align*}
            G(y) &= (y+y^2+...)(y^2+y^3+...)(y^3+y^4+...)...(y^{n}+y^{n+1}+...)\\
            &= \prod_{i=1}^n (y^i \times Y) = Y^n \times \prod_{i=1}^n y^i = Y^n \times y^{\sum_{i=1}^n i} = Y^n \times y^{\frac{n(n+1)}{2}}
        \end{align*}
        طبق \CROSSREF{جدول توابع مولد پرکاربرد}،
        داریم:
        $$Y = (1-y)^{-1}$$
        پس:
        $$G(y) = (1-y)^{-n} \times y^{\frac{n(n+1)}{2}}$$
        پاسخ مسئله، برابر است با ضریب
        $y^k$
        در بسط $G(y)$
        یا همان ضریب
        $y^{k-\frac{n(n+1)}{2}}$
        در بسط $(1-y)^{-n}$.
        مجددا طبق جدول توابع مولد پرکاربرد، داریم:
        $$(1-y)^{-n} = \sum_{i=1}^\infty {n+i-1 \choose i} y^i$$
        بنابراین، پاسخ مسئله برابر است با:
        $${n+{k-\frac{n(n+1)}{2}}-1 \choose {k-\frac{n(n+1)}{2}}} = {n+{k-\frac{n(n+1)}{2}}-1 \choose n-1}$$
    }
\end{PROBLEM}
