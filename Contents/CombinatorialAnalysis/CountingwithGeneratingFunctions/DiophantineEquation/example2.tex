\begin{PROBLEM}
    \p
    تعداد جواب‌های معادله زیر را در مجموعه اعداد حسابی بیابید به طوری که:
    $$2w + 3y + 7z = r$$

    \begin{enumerate}
        \item 
        $$w,y,z \ge 0, r=12$$

        \SOLUTION{
            \p
            پاسخ مسئله برابر است با ضریب $x^{12}$
            در عبارت زیر:
            $$\underbrace{(1+x^2+x^4+x^6+...)}_{2w}\underbrace{(1+x^3+x^6+x^9+...)}_{3y}\underbrace{(1+x^7+x^{14}+...)}_{7z}$$
            چرا که ضریب $x^{12}$
            درواقع تعداد دفعاتی است که
            $e_1+e_2+e_3=12$
            که
            $$e_1 = 2w \in {0,2,4,...}$$
            $$e_2 = 3y \in {0,3,6,...}$$
            $$e_3 = 7z \in {0,7,14,...}$$
            \p
            مشخص است که عبارت بالا معادل معادله سیاله داده شده است.
            با بسط تابع مولد نوشته شده، مشاهده می‌شود که ضریب $x^{12}$
            و درنتیجه پاسخ مسئله برابر است با
            $4$.
        }

        \item 
        $$0 \le z \le 2 \le y \le 8 \le w, r=46$$

        \SOLUTION{
            \p
            پاسخ مسئله برابر است با ضریب $x^{46}$
            در عبارت زیر:
            $$\underbrace{(x^{16}+x^{18}+x^{20}+...)}_{2w}\underbrace{(x^6+x^9+...+x^{24})}_{3y}\underbrace{(1+x^7+x^{14})}_{7z}$$
            با بسط تابع مولد نوشته شده، مشاهده می‌شود که ضریب $x^{46}$
            و درنتیجه پاسخ مسئله برابر است با
            $9$.
        }
    \end{enumerate}
\end{PROBLEM}
