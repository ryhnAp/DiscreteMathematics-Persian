\SUBSECTION{اتحاد پاسکال}

\begin{PROBLEM}[اتحاد پاسکال]
  \p
  اگر
  $n,j$
  اعداد صحیح مثبت باشند و
  $n \geq j$،
  اثبات کنید:
  $$\binom{n-1}{j} + \binom{n-1}{j-1} = \binom{n}{j}$$

  \SOLUTION{
    \p
    برای اثبات این اتحاد از 
    \CROSSREF[دوگانه شماری]{توضیح روش ترکیبیاتی دوگانه شماری}
    استفاده می‌کنیم.
    فرض کنید جمعیتی
    $n$
    نفری داریم که یکی از آنان رهبر است.
    قصد داریم یک نمونه
    $j$
    نفری از این جمعیت انتخاب کنیم که حضور یا عدم حضور رهبر در آن بی‌اهمیت است.
    هدف یافتن تعداد حالات انتخاب این نمونه است.
    این مسئله به دو طریق قابل حل است:
    \begin{enumerate}
      \item 
      $j$
      نفر از بین
      $n$
      نفر انتخاب می‌کنیم:
        $$\binom{n}{j}$$

      \item 
      دو حالت برای نمونه برداری درنظر می‌گیریم:
      \begin{enumerate}
        \item 
        رهبر عضوی از نمونه باشد که در این صورت، انتخاب یک نفر قطعی بوده
        و باید مابین
        $n-1$
        نفر باقیمانده
        $j-1$
        نفر انتخاب شود:
          $$\binom{n-1}{j-1}$$

        \item 
        رهبر عضو نمونه نباشد که در این صورت، عدم انتخاب یک نفر قطی بوده
        و باید مابین
        $n-1$
        نفر باقیمانده
        $j$
        نفر انتخاب شود:
          $$\binom{n-1}{j}$$
      \end{enumerate}
      طبق اصل جمع، تعداد حالات این طریق نمونه برداری برابر است با:
        $$\binom{n-1}{j-1} + \binom{n-1}{j}$$
    \end{enumerate}
    \p
    با توجه به یکتا بودن پاسخ مسئله، پاسخ هر دو نوع شمارش برابر است. پس:
      $$\binom{n-1}{j} + \binom{n-1}{j-1} = \binom{n}{j}$$
  }
\end{PROBLEM}

\begin{THEOREM}
  \TARGET{اتحاد پاسکال برای اعداد حسابی}
  \p
    \FOCUSEDON{اتحاد پاسکال:}
    اگر
    $n,j$
    اعداد صحیح مثبت باشند و
    $n \geq j$،
    داریم:
    $$\binom{n-1}{j} + \binom{n-1}{j-1} = \binom{n}{j}$$
\end{THEOREM}
