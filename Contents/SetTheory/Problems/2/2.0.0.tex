ب)
\begin{align*}
    A(x) &= 1 + x + x^2 + ... = \frac{1}{1 - x}\\
    A(x^2) &= 1 + x^2 + x^4 + ... = \frac{1}{1 - x^2}\\
    2A(x^2) &= 2 + 2x^2 + 2x^4 + ... = \frac{2}{1 - x^2}     
\end{align*}

ج)
\begin{align*}
    A(x) &= 1 + x + x^2 + ... = \frac{1}{1 - x}\\
    A^2(x) &= 1 + 2x + 3x^2 + ... = \frac{1}{(1 - x)^2}\\ 
    A^2(x) - 1 &= 2x + 3x^2 + 4x^3 + ... = \frac{1}{(1 - x)^2} - 1\\
    \frac{A^2(x) - 1}{x} &= 2 + 3x + 4x^2 + ... = \frac{\frac{1}{(1 - x)^2} - 1}{x}
\end{align*}
